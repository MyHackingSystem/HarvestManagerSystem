//*******************************
        //Add new Supplier data
        //*******************************
        internal bool addNewSupplierData(HarvestHours harvestHours)
        {
            string insertStmt = "INSERT INTO HarvestHours_For_Insert" +
                                "(HarvestDate, TimeStartMorning,"+
                                "TimeEndMorning, TimeStartNoon, TimeEndNoon, " +
                                "HourPrice, Remarque, EmployeeType, EmployeeId, " +
                                "TransportDate, TransportAmount, EmployeeId," +
                                "FarmId, CreditDate, CreditAmount, EmployeeId, ProductionId ) VALUES ("
                                + "@HarvestDate, "
                                + "@TimeStartMorning, "
                                + "@TimeEndMorning, "
                                + "@TimeStartNoon, "
                                + "@TimeEndNoon, "
                                + "@HourPrice, "
                                + "@Remarque, "
                                + "@EmployeeType, "
                                + "@EmployeeId, "
                                + "@TransportDate, "
                                + "@TransportAmount, "
                                + "@EmployeeId, "
                                + "@FarmId, "
                                + "@CreditDate, "
                                + "@CreditAmount, "
                                + "@EmployeeId, "
                                + "@ProductionId "
                                + " )";
            try
            {
                SQLiteCommand sQLiteCommand = new SQLiteCommand(insertStmt, mSQLiteConnection);
                OpenConnection();
                sQLiteCommand.Parameters.AddWithValue("@HarvestDate", harvestHours.HarvestDate);
                sQLiteCommand.Parameters.AddWithValue("@TimeStartMorning", harvestHours.TimeStartMorning);
                sQLiteCommand.Parameters.AddWithValue("@TimeEndMorning", harvestHours.TimeEndMorning);
                sQLiteCommand.Parameters.AddWithValue("@TimeStartNoon", harvestHours.TimeStartNoon);
                sQLiteCommand.Parameters.AddWithValue("@TimeEndNoon", harvestHours.TimeEndNoon);
                sQLiteCommand.Parameters.AddWithValue("@HourPrice", harvestHours.HourPrice);
                sQLiteCommand.Parameters.AddWithValue("@Remarque", harvestHours.Remarque);
                sQLiteCommand.Parameters.AddWithValue("@EmployeeType", harvestHours.EmployeeType);
                sQLiteCommand.Parameters.AddWithValue("@EmployeeId", harvestHours.Employee.EmployeeId);
                sQLiteCommand.Parameters.AddWithValue("@TransportDate", harvestHours.HarvestDate);
                sQLiteCommand.Parameters.AddWithValue("@TransportAmount", harvestHours.Transport.TransportAmount);
                sQLiteCommand.Parameters.AddWithValue("@EmployeeId", harvestHours.Employee.EmployeeId);
                sQLiteCommand.Parameters.AddWithValue("@FarmId", harvestHours.Production.Farm.FarmId);
                sQLiteCommand.Parameters.AddWithValue("@CreditDate", harvestHours.HarvestDate);
                sQLiteCommand.Parameters.AddWithValue("@CreditAmount", harvestHours.Credit.CreditAmount);
                sQLiteCommand.Parameters.AddWithValue("@EmployeeId", harvestHours.Employee.EmployeeId);
                sQLiteCommand.Parameters.AddWithValue("@ProductionId", harvestHours.Production.ProductionID);
                sQLiteCommand.ExecuteNonQuery();
                return true;
            }
            catch (SQLiteException e)
            {
                Console.WriteLine(e.Message);
                return false;
            }
            finally
            {
                CloseConnection();
            }
        }
